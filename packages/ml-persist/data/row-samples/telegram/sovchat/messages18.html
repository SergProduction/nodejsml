function findCommonDenominators(rationals = '') {
  const numerators = [];
  const denumerators = [];

  function reduceFraction(numerator, denumerator) {
    if (!(denumerator / numerator === Math.ceil(denumerator / numerator))) {
      const commonNumber = Number(
        String(denumerator / numerator).split('.')[0],
      );

      if (
        numerator / commonNumber === Math.ceil(numerator / commonNumber) &&
        denumerator / commonNumber === Math.ceil(denumerator / commonNumber)
      ) {
        return [numerator / commonNumber, denumerator / commonNumber];
      }

      return [numerator, denumerator];
    }

    return [numerator / numerator, denumerator / numerator];
  }

  rationals.split(',').forEach((num, index) => {
    const isOdd = (index + 1) % 2 === 1;

    if (isOdd) {
      numerators.push(Number(num));
    } else {
      denumerators.push(Number(num));
    }
  });

  for (let i = 0; i < numerators.length; i++) {
    [numerators[i], denumerators[i]] = reduceFraction(
      numerators[i],
      denumerators[i],
    );
